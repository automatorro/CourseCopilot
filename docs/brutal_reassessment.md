function validateWorkbook(workbook: string, blueprint: any): ValidationResult {
  const errors = [];
  const warnings = [];
  
  // Check 1: Word count (should be 12,000-15,000)
  const wordCount = workbook.split(/\s+/).length;
  if (wordCount < 10000) {
    errors.push(`Workbook too short: ${wordCount} words (need 12,000+)`);
  }
  
  // Check 2: All modules present
  const moduleHeaders = workbook.match(/## Module \d+:/g) || [];
  if (moduleHeaders.length !== blueprint.modules.length) {
    errors.push(`Missing modules: found ${moduleHeaders.length}, expected ${blueprint.modules.length}`);
  }
  
  // Check 3: Exercises present
  const exerciseCount = (workbook.match(/ğŸ¯ EXERCIÈšIU/g) || []).length;
  const expectedExercises = calculateExpectedExercises(blueprint);
  if (exerciseCount < expectedExercises * 0.8) { // Allow 20% margin
    warnings.push(`Few exercises: ${exerciseCount} found, expected ~${expectedExercises}`);
  }
  
  // Check 4: Not just bullets (should have paragraphs)
  const bulletLines = (workbook.match(/^\s*[-*â€¢]/gm) || []).length;
  const totalLines = workbook.split('\n').length;
  if (bulletLines / totalLines > 0.5) {
    warnings.push(`Too many bullets: ${Math.round(bulletLines/totalLines*100)}% of lines. Should have more prose.`);
  }
  
  return {
    passed: errors.length === 0,
    errors,
    warnings
  };
}
